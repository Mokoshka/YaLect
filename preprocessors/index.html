Миксины


$boxes = 'item1', 'item2', 'item3';
set_width($len) {
    if $len >= 3 {
        width: 300px;
    } else {
        width: 200px;
    }
}
.box {
    set-width(length($boxes))
}
            

.box {
    width: 300px;
}
